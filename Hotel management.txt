single_rooms = 10
double_rooms = 5

print("1. Single Room")
print("2. Double Room")
print("3. Suite Room")

choice = int(input("Enter your option (1-3): "))

if choice == 1:
    if single_rooms > 0:
        print("Single Room booked successfully!")
        single_rooms -= 1
        print("Remaining Single Rooms:", single_rooms)
    else:
        print("Sorry, no Single Rooms available.")

elif choice == 2:
    if double_rooms > 0:
        print("Double Room booked successfully!")
        double_rooms -= 1
        print("Remaining Double Rooms:", double_rooms)
    else:
        print("Sorry, no Double Rooms available.")

elif choice == 3:
    if suite_rooms > 0:
        print("Suite Room booked successfully!")
        suite_rooms -= 1
        print("Remaining Suite Rooms:", suite_rooms)
    else:
        print("Sorry, no Suite Rooms available.")

else:
    print("Invalid choice... Please select 1, 2, or 3.")
    exit()
